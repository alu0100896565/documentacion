def each(&block)
  return nil if @head == nil
  nodo = @head
  while true
  block.call(nodo.value)
  #yield nodo.value
  if nodo.next == nil
    break
  end
  nodo = nodo.next
  
  end
end

def extract_head()
  
  if self.empty
    puts "La lista esta vacia"
    return nil
  end
  nodo = @head
  @head = @head.next
  if @head != nil
    @head.prev = nil;
  else @tail = nil
  end
  nodo.prev = nil
  nodo.next = nil
  return nodo.value
end

def insert_head(val)
    nodo = Node.new(val,nil,nil)
  if self.empty
    @head = nodo
    @tail = nodo
  else 
    @head.prev = nodo
    nodo.next = @head
    @head = nodo
  end
end

def recorrerLista
  
  nodo=@head
  content = ""
  while nodo!=nil do
    content += " " + nodo.value.to_s
    nodo = nodo.next
  end
  return content
end
